import java.util.Scanner;

public class Introduccion {

	public static void main(String[] args) {
		float imc;
		float peso;
		float altura;
		Scanner leer = new Scanner(System.in);
		String diagnostico;
		
		System.out.println("Coloque su peso en kg:");
		peso = leer.nextFloat();
		System.out.println("Coloque su altura en metros:");
		altura = leer.nextFloat();
		
		imc = (peso/altura);
		
		if(imc < 20)
			diagnostico = "Mal estado";
		else if((imc >= 20) && (imc <= 22))
			diagnostico = "Bajo de peso";
		else if((imc >= 22) && (imc <= 25))
			diagnostico = "Peso normal";
		else if((imc >= 25) && (imc <= 30))
			diagnostico = "Sobrepeso";
		else if ((imc >= 30) && (imc <= 40))
			diagnostico = "Sobrepeso cronico";
		else
			diagnostico = "Hospitalizado";
		
		System.out.println("Su diagnostico es: "+diagnostico);
		

	}

}
